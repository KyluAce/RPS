\subsection{Zadanie 15}

$$
\phi(x) =
\left\{ \begin{array}{ll}
1 & \text{gdy } x \in [\frac{1}{2},1]\\
0 & \text{gdy } x \notin [\frac{1}{2},1]
\end{array} \right.
$$

$$
Y = \phi(X)
$$

$$
P(Y = 1) = P(X = \phi^{-1}(1)) = P(X \in [\frac{1}{2},1]) = \frac{1}{2}
$$

$$
P(Y = 0) = P(\phi(X) = 0) = P(X = \phi^{-1}(0)) = P(X \in [0,\frac{1}{2}]) = \frac{1}{2}
$$
